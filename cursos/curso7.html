Introducción a la Programación en C

Bienvenido al curso completo de programación en C. C es un lenguaje de programación de propósito general que ha influido en muchos otros lenguajes modernos. Es conocido por su eficiencia y control sobre los recursos del sistema.

En este curso aprenderás desde los conceptos más básicos hasta temas avanzados, con ejemplos prácticos que podrás probar directamente en esta página.

¿Por qué aprender C?

Eficiencia

C permite un control detallado sobre los recursos del sistema, lo que resulta en programas muy eficientes.

Portabilidad

El código C puede compilarse en casi cualquier plataforma sin cambios significativos.

Base para otros lenguajes

Muchos lenguajes modernos como C++, Java y C# están influenciados por C.

Ampliamente utilizado

C se usa en sistemas operativos, compiladores, bases de datos y aplicaciones de tiempo real.

Conceptos Básicos de C

Estos son los fundamentos de la programación en C. Comprender estos conceptos es esencial para avanzar.

Ejercicio 8: Calculadora simple
#include <stdio.h>

int main() {
    char operador;
    float num1, num2, resultado;
    
    printf("Ingresa una operación (+, -, *, /): ");
    scanf(" %c", &operador);
    
    printf("Ingresa dos números: ");
    scanf("%f %f", &num1, &num2);
    
    switch(operador) {
        case '+':
            resultado = num1 + num2;
            break;
        case '-':
            resultado = num1 - num2;
            break;
        case '*':
            resultado = num1 * num2;
            break;
        case '/':
            if (num2 != 0) {
                resultado = num1 / num2;
            } else {
                printf("Error: No se puede dividir por cero\\n");
                return 1;
            }
            break;
        default:
            printf("Operador no válido\\n");
            return 1;
    }
    
    printf("Resultado: %.2f %c %.2f = %.2f\\n", num1, operador, num2, resultado);
    
    return 0;
}

Las funciones permiten dividir el código en partes más pequeñas y reutilizables, mejorando la organización y legibilidad.

Ejercicios con funciones básicas

Ejercicio 15: Función para calcular potencia
#include <stdio.h>

// Declaración de la función
double potencia(double base, int exponente);

int main() {
    double base, resultado;
    int exponente;
    
    printf("Ingresa la base: ");
    scanf("%lf", &base);
    
    printf("Ingresa el exponente: ");
    scanf("%d", &exponente);
    
    resultado = potencia(base, exponente);
    
    printf("%.2f elevado a %d es: %.2f\\n", base, exponente, resultado);
    
    return 0;
}

// Definición de la función
double potencia(double base, int exponente) {
    double resultado = 1.0;
    int i;
    
    for (i = 0; i < exponente; i++) {
        resultado *= base;
    }
    
    return resultado;
}

Ejercicios con funciones recursivas

Ejercicio 17: Factorial usando recursión
#include <stdio.h>

// Declaración de la función recursiva
long long factorial(int n);

int main() {
    int numero;
    
    printf("Ingresa un número para calcular su factorial: ");
    scanf("%d", &numero);
    
    if (numero < 0) {
        printf("Error: El factorial no está definido para números negativos.\\n");
    } else {
        printf("El factorial de %d es: %lld\\n", numero, factorial(numero));
    }
    
    return 0;
}

// Definición de la función recursiva
long long factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}
Ejercicio 18: Fibonacci usando recursión
#include <stdio.h>

// Declaración de la función recursiva
long long fibonacci(int n);

int main() {
    int n, i;
    
    printf("Ingresa el número de términos de Fibonacci: ");
    scanf("%d", &n);
    
    printf("Serie de Fibonacci:\\n");
    for (i = 0; i < n; i++) {
        printf("%lld", fibonacci(i));
        if (i < n - 1) {
            printf(", ");
        }
    }
    printf("\\n");
    
    return 0;
}

// Definición de la función recursiva
long long fibonacci(int n) {
    if (n <= 1) {
        return n;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

Arreglos en C

Los arreglos permiten almacenar múltiples valores del mismo tipo en una sola variable, facilitando el manejo de colecciones de datos.

Ejercicios con arreglos unidimensionales

Ejercicio 19: Promedio de calificaciones
#include <stdio.h>

#define MAX 100

int main() {
    int calificaciones[MAX];
    int n, i;
    float suma = 0, promedio;
    
    printf("Ingresa el número de calificaciones: ");
    scanf("%d", &n);
    
    if (n > MAX || n <= 0) {
        printf("Número de calificaciones no válido.\\n");
        return 1;
    }
    
    printf("Ingresa las calificaciones:\\n");
    for (i = 0; i < n; i++) {
        printf("Calificación %d: ", i + 1);
        scanf("%d", &calificaciones[i]);
        suma += calificaciones[i];
    }
    
    promedio = suma / n;
    
    printf("El promedio de las calificaciones es: %.2f\\n", promedio);
    
    return 0;
}
Ejercicio 20: Encontrar el máximo y mínimo en un arreglo
#include <stdio.h>

#define MAX 100

int main() {
    int arr[MAX];
    int n, i;
    int max, min;
    
    printf("Ingresa el número de elementos: ");
    scanf("%d", &n);
    
    if (n > MAX || n <= 0) {
        printf("Número de elementos no válido.\\n");
        return 1;
    }
    
    printf("Ingresa los elementos del arreglo:\\n");
    for (i = 0; i < n; i++) {
        printf("Elemento %d: ", i + 1);
        scanf("%d", &arr[i]);
    }
    
    max = min = arr[0];
    
    for (i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
        if (arr[i] < min) {
            min = arr[i];
        }
    }
    
    printf("El valor máximo es: %d\\n", max);
    printf("El valor mínimo es: %d\\n", min);
    
    return 0;
}

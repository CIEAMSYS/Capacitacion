/* Conversión Fahrenheit-Celsius (entero) */
#include 
int main(void) {
    int fahr, celsius;
    int lower = 0, upper = 300, step = 20;
    fahr = lower;
    while (fahr <= upper) {
        celsius = 5 * (fahr - 32) / 9;
        printf("%d\\t%d\\n", fahr, celsius);
        fahr = fahr + step;
    }
    return 0;
}

/* Potencia con for */
#include 
int power(int base, int n) {
    int p = 1;
    for (int i = 1; i <= n; ++i)
        p *= base;
    return p;
}
int main(void) {
    for (int i = 0; i < 10; ++i)
        printf("%d %d %d\\n", i, power(2,i), power(-3,i));
    return 0;
}

/* Función que imprime encabezado y tabla de Fahrenheit a Celsius */
#include 
float fahr2celsius(float fahr) {
    return (5.0/9.0)*(fahr-32.0);
}
int main(void) {
    float fahr;
    int lower=0,upper=300,step=20;
    for (fahr=lower; fahr<=upper; fahr+=step)
        printf("%3.0f %6.1f\\n", fahr, fahr2celsius(fahr));
    return 0;
}

/* Uso de apuntadores: strcpy */
#include 
void mystrcpy(char *s, char *t) {
    while ((*s++ = *t++) != '\0');
}
int main(void) {
    char origen[] = "Hola";
    char destino[10];
    mystrcpy(destino, origen);
    printf("%s\\n", destino);
    return 0;
}

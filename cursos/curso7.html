Capítulo 2: Variables y Tipos de Datos

      
En C, las variables son contenedores para almacenar datos. Cada variable tiene un tipo específico.

      
      
// Ejemplo 2.1: Tipos de datos básicos
#include <stdio.h>

int main() {
    int entero = 10;
    float decimal = 3.14;
    double doble = 3.141592;
    char caracter = 'A';
    char cadena[] = "Hola C";
    
    printf("Entero: %d\n", entero);
    printf("Float: %.2f\n", decimal);
    printf("Double: %.6f\n", doble);
    printf("Caracter: %c\n", caracter);
    printf("Cadena: %s\n", cadena);
    
    return 0;
}

      

      
// Ejemplo 2.2: Constantes y operaciones
#include <stdio.h>
#define PI 3.14159

int main() {
    const int DIAS_SEMANA = 7;
    float radio = 5.0;
    float area = PI * radio * radio;
    
    printf("Días en una semana: %d\n", DIAS_SEMANA);
    printf("Área del círculo (radio=%.1f): %.2f\n", radio, area);
    
    return 0;
}

      
    

    
    
    

      
Capítulo 3: Estructuras de Control

      
Las estructuras de control permiten dirigir el flujo de ejecución de un programa.

      
      
// Ejemplo 3.1: Condicional if-else
#include <stdio.h>

int main() {
    int edad = 18;
    
    if(edad >= 18) {
        printf("Eres mayor de edad\n");
    } else {
        printf("Eres menor de edad\n");
    }
    
    // Operador ternario
    char* mensaje = (edad >= 18) ? "Mayor" : "Menor";
    printf("Estado: %s\n", mensaje);
    
    return 0;
}

      

      
// Ejemplo 3.2: Bucle for y while
#include <stdio.h>

int main() {
    int i;
    
    printf("Bucle for:\n");
    for(i = 1; i <= 5; i++) {
        printf("Número: %d\n", i);
    }
    
    printf("\nBucle while:\n");
    int contador = 5;
    while(contador > 0) {
        printf("Contador: %d\n", contador);
        contador--;
    }
    
    return 0;
}

      

      
// Ejemplo 3.3: Switch case
#include <stdio.h>

int main() {
    int opcion = 2;
    
    switch(opcion) {
        case 1:
            printf("Opción 1 seleccionada\n");
            break;
        case 2:
            printf("Opción 2 seleccionada\n");
            break;
        case 3:
            printf("Opción 3 seleccionada\n");
            break;
        default:
            printf("Opción no válida\n");
    }
    
    return 0;
}

      
    

    
    
    

      
Capítulo 4: Funciones

      
Las funciones permiten dividir un programa en partes más pequeñas y manejables.

      
      
// Ejemplo 4.1: Función básica
#include <stdio.h>

// Declaración de función
int sumar(int a, int b);

int main() {
    int resultado = sumar(8, 12);
    printf("La suma es: %d\n", resultado);
    return 0;
}

// Definición de función
int sumar(int a, int b) {
    return a + b;
}

      

      
// Ejemplo 4.2: Función recursiva (factorial)
#include <stdio.h>

long factorial(int n);

int main() {
    int numero = 5;
    long fact = factorial(numero);
    printf("El factorial de %d es %ld\n", numero, fact);
    return 0;
}

long factorial(int n) {
    if(n <= 1) return 1;
    return n * factorial(n - 1);
}

      

      
// Ejemplo 4.3: Función con múltiples parámetros
#include <stdio.h>

void imprimirInfo(char nombre[], int edad, float altura);

int main() {
    imprimirInfo("Ana", 25, 1.68);
    imprimirInfo("Carlos", 30, 1.75);
    return 0;
}

void imprimirInfo(char nombre[], int edad, float altura) {
    printf("Nombre: %s, Edad: %d, Altura: %.2f\n", nombre, edad, altura);
}

      
    

    
    
    

      
Capítulo 5: Arreglos y Cadenas

      
Los arreglos son estructuras de datos que almacenan múltiples valores del mismo tipo.

      
      
// Ejemplo 5.1: Arreglo de enteros
#include <stdio.h>

int main() {
    int numeros[5] = {10, 20, 30, 40, 50};
    int i;
    
    printf("Elementos del arreglo:\n");
    for(i = 0; i < 5; i++) {
        printf("numeros[%d] = %d\n", i, numeros[i]);
    }
    
    printf("\nSuma de elementos: ");
    int suma = 0;
    for(i = 0; i < 5; i++) {
        suma += numeros[i];
    }
    printf("%d\n", suma);
    
    return 0;
}

      

      
// Ejemplo 5.2: Manejo de cadenas
#include <stdio.h>
#include <string.h>

int main() {
    char saludo[20] = "Hola";
    char nombre[] = "Mundo";
    char mensaje[50];
    
    // Concatenar cadenas
    strcpy(mensaje, saludo);
    strcat(mensaje, " ");
    strcat(mensaje, nombre);
    strcat(mensaje, "!");
    
    printf("Cadena resultante: %s\n", mensaje);
    printf("Longitud: %zu\n", strlen(mensaje));
    
    // Comparar cadenas
    if(strcmp(saludo, "Hola") == 0) {
        printf("El saludo es 'Hola'\n");
    }
    
    return 0;
}

      

      
// Ejemplo 5.3: Arreglo bidimensional (matriz)
#include <stdio.h>

int main() {
    int matriz[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    
    int i, j;
    printf("Matriz 3x3:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            printf("%d ", matriz[i][j]);
        }
        printf("\n");
    }
    
    return 0;
}

      
    

    
    
    

      
Capítulo 6: Punteros

      
Los punteros son variables que almacenan direcciones de memoria.

      
      
// Ejemplo 6.1: Punteros básicos
#include <stdio.h>

int main() {
    int numero = 42;
    int *p = №
    
    printf("Valor de numero: %d\n", numero);
    printf("Dirección de numero: %p\n", &numero);
    printf("Valor del puntero p: %p\n", p);
    printf("Valor al que apunta p: %d\n", *p);
    
    // Modificar valor a través del puntero
    *p = 100;
    printf("Nuevo valor de numero: %d\n", numero);
    
    return 0;
}

      

      
// Ejemplo 6.2: Punteros y arreglos
#include <stdio.h>

int main() {
    int arr[5] = {1, 2, 3, 4, 5};
    int *ptr = arr;  // ptr apunta al primer elemento
    
    printf("Acceso mediante puntero:\n");
    for(int i = 0; i < 5; i++) {
        printf("arr[%d] = %d, *(ptr + %d) = %d\n", i, arr[i], i, *(ptr + i));
    }
    
    printf("\nUsando aritmética de punteros:\n");
    for(ptr = arr; ptr < arr + 5; ptr++) {
        printf("%d ", *ptr);
    }
    printf("\n");
    
    return 0;
}

      

      
// Ejemplo 6.3: Punteros a funciones
#include <stdio.h>

int sumar(int a, int b) {
    return a + b;
}

int multiplicar(int a, int b) {
    return a * b;
}

int main() {
    int (*operacion)(int, int);  // Puntero a función
    
    operacion = sumar;
    printf("Suma: %d\n", operacion(5, 3));
    
    operacion = multiplicar;
    printf("Multiplicación: %d\n", operacion(5, 3));
    
    return 0;
}

      
    

    
    
    

      
Capítulo 7: Estructuras y Uniones

      
Las estructuras permiten agrupar variables de diferentes tipos bajo un mismo nombre.

      
      
// Ejemplo 7.1: Estructura básica
#include <stdio.h>
#include <string.h>

// Definición de estructura
struct Estudiante {
    char nombre[50];
    int edad;
    float promedio;
};

int main() {
    // Declaración e inicialización
    struct Estudiante estudiante1;
    strcpy(estudiante1.nombre, "María López");
    estudiante1.edad = 20;
    estudiante1.promedio = 8.5;
    
    printf("Estudiante:\n");
    printf("Nombre: %s\n", estudiante1.nombre);
    printf("Edad: %d\n", estudiante1.edad);
    printf("Promedio: %.1f\n", estudiante1.promedio);
    
    return 0;
}

      

      
// Ejemplo 7.2: Arreglo de estructuras
#include <stdio.h>

struct Punto {
    int x;
    int y;
};

int main() {
    struct Punto puntos[3] = {
        {1, 2},
        {3, 4},
        {5, 6}
    };
    
    printf("Coordenadas de puntos:\n");
    for(int i = 0; i < 3; i++) {
        printf("Punto %d: (%d, %d)\n", i+1, puntos[i].x, puntos[i].y);
    }
    
    return 0;
}

      

      
// Ejemplo 7.3: Uniones
#include <stdio.h>

union Dato {
    int entero;
    float decimal;
    char caracter;
};

int main() {
    union Dato dato;
    
    dato.entero = 10;
    printf("Valor entero: %d\n", dato.entero);
    
    dato.decimal = 3.14;
    printf("Valor decimal: %.2f\n", dato.decimal);
    
    dato.caracter = 'A';
    printf("Valor caracter: %c\n", dato.caracter);
    
    // Solo el último valor asignado es válido
    printf("Entero después de cambiar: %d (valor incorrecto)\n", dato.entero);
    
    return 0;
}
